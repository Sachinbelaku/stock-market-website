        // Function to create a stock card and add it to the page

        function createStockCard(stock) {

            const card = document.createElement('div');

            card.className = 'bg-white p-6 rounded-2xl shadow-md transition-all duration-300 transform hover:scale-105 hover:shadow-xl cursor-pointer border border-gray-200';

            card.onclick = () => showModal(stock);


            const changeColor = stock.changeType === 'positive' ? 'text-green-500' : 'text-red-500';


            card.innerHTML = `

                <h3 class="text-xl font-semibold text-gray-800 mb-1">${stock.name}</h3>

                <p class="text-sm text-gray-500 mb-2">${stock.symbol} | <span class="${changeColor} font-medium">${stock.change}</span></p>

                <p class="text-2xl font-bold text-gray-900">${stock.price}</p>

            `;

            return card;

        }


        // Loop through the stocks array and add each card to the page

        const stockListContainer = document.getElementById('stock-list');

        stocks.forEach(stock => {

            stockListContainer.appendChild(createStockCard(stock));

        // Get modal elements

        const modal = document.getElementById('stock-modal');

        const modalContent = modal.querySelector('.modal-content');

        const modalTitle = document.getElementById('modal-title');

        const modalDescription = document.getElementById('modal-description');

        const modalPrice = document.getElementById('modal-price');

        const modalChange = document.getElementById('modal-change');


        // Function to show the modal with stock details

        function showModal(stock) {

            modalTitle.textContent = stock.name;

            modalDescription.textContent = stock.description;

            modalPrice.textContent = `Current Price: ${stock.price}`;

            

            const changeColor = stock.changeType === 'positive' ? 'text-green-600' : 'text-red-600';

            modalChange.innerHTML = `Change: <span class="${changeColor}">${stock.change}</span>`;


            modal.classList.remove('hidden');

            modal.classList.add('flex');

            setTimeout(() => {

                modalContent.classList.remove('scale-95', 'opacity-0');

                modalContent.classList.add('scale-100', 'opacity-100');

            }, 10);

        }


        // Function to hide the modal

        function closeModal() {

            modalContent.classList.remove('scale-100', 'opacity-100');

            modalContent.classList.add('scale-95', 'opacity-0');

            setTimeout(() => {

                modal.classList.remove('flex');

                modal.classList.add('hidden');

            }, 300);

        }


        // Close modal when clicking outside of it

        modal.addEventListener('click', (e) => {

            if (e.target === modal) {

                closeModal();

            }
